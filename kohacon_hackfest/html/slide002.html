use Template::Plugin::Filter;
use base qw( Template::Plugin::Filter );
use warnings;
use strict;

use C4::Dates;

sub filter {
   my ($self,$text) = @_;
   return "" if not $text;
   my $date = C4::Dates->new( $text, 'iso' );
   return $date->output("syspref");
}
